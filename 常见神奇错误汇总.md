```c++
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<vector>
#include<queue>
#include<map>
using namespace std;

int w[210][210],dp[210][50];
int p,k,ss;

int main(){
    string temp,s;
    cin>>p>>k;
    s+='0';
    for(int i=0;i<p;i++){
        cin>>temp;
        s+=temp;
    }
    cin>>ss;
    string dict[ss+10];
    for(int i=0;i<ss;i++){
        cin>>dict[i];
    }

    for(int j=s.length()-1;j>=1;j--){
        for(int i=j;i>=1;i--){
            w[i][j]=w[i+1][j];
            for(int p=0;p<ss;p++){
                if(s.substr(i,j-i+1).find(dict[p])==0){
                    w[i][j]++;
                    break;
                }
            }
            
        }
    }
    
    for(int i=1;i<=k;i++){
        dp[i][i]=dp[i-1][i-1]+w[i][i];
    }
    for(int i=1;i<=s.length()-1;i++){
        dp[i][1]=w[1][i];
    }
    for(int i=1;i<=s.length()-1;i++){
        for(int j=1;j<=k&&j<i;j++){
            for(int l=j;l<i;l++){
                dp[i][j]=max(dp[i][j],dp[l][j-1]+w[l+1][i]);
            }
        }
    }
    cout<<dp[s.length()-1][k];
}
```
